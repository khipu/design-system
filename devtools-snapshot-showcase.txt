    isDrawing = true;
    ctx.beginPath();

    if (!isDrawing) return;
    ctx.lineWidth = 2;
    ctx.lineCap = 'round';
    ctx.strokeStyle = '#111827';
    ctx.lineTo(e.offsetX, e.offsetY);
    ctx.stroke();

canvas.addEventListener('mouseup', () => isDrawing = false);
canvas.addEventListener('mouseout', () => isDrawing = false);

    const rect = canvas.getBoundingClientRect();
