Card Selector

Tarjetas seleccionables con icono, título y descripción

person
Persona Natural
Emprendedor individual o freelancer
business
Persona Jurídica
Empresa o sociedad legalmente constituida 
<div class="kds-grid-2col">
    <div class="kds-card-selector" data-value="natural" onclick="selectCard(this)">
        <div class="kds-card-selector-icon">
            <i class="material-symbols-outlined">person</i>
        </div>
        <div class="kds-card-selector-title">Persona Natural</div>
        <div class="kds-card-selector-description">
            Emprendedor individual o freelancer
        </div>
    </div>
</div>

<script>
function selectCard(card) {
    card.parentElement.querySelectorAll('.kds-card-selector').forEach(c =>
        c.classList.remove('selected')
    );
    card.classList.add('selected');
}
</script>

Usage Notes

  • Grids: .kds-grid-2col (2 columnas), .kds-grid-3col (3 columnas)
  • Selection: Clase .selected muestra borde azul
  • Responsive: Colapsa a 1 columna en mobile

Grails/GSP Integration

<!-- Selector de tipo de persona -->
<div class="kds-grid-2col">
    <g:each in="${personTypes}" var="type">
        <div class="kds-card-selector ${type.value == selectedType ? 'selected' : ''}"
             data-value="${type.value}"
             onclick="selectCard(this)">
            <div class="kds-card-selector-icon">
                <i class="material-symbols-outlined">${type.icon}</i>
            </div>
            <div class="kds-card-selector-title">${type.title}</div>
            <div class="kds-card-selector-description">
                ${type.description}
            </div>
        </div>
    </g:each>
</div>

<!-- Hidden field para enviar valor seleccionado -->
<g:hiddenField name="personType" value="${selectedType}"/>

Grails/GSP Integration

<!-- Grid de selección de bancos -->
<div class="kds-bank-grid">
    <g:each in="${availableBanks}" var="bank">
        <div class="kds-bank-item ${bank.code == selectedBank ? 'selected' : ''}"
             data-bank="${bank.code}"
             onclick="selectBank(this)">
            <asset:image src="banks/${bank.code}.png" alt="${bank.name}"/>
        </div>
    </g:each>
</div>

<!-- Hidden field para el banco seleccionado -->
<g:hiddenField name="bankCode" value="${selectedBank}" id="bankCodeField"/>

<script>
function selectBank(item) {
    document.querySelectorAll('.kds-bank-item').forEach(b => b.classList.remove('selected'));
    item.classList.add('selected');
    document.getElementById('bankCodeField').value = item.dataset.bank;
}
</script>
