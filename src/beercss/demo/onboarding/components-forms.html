Usage Notes (Material Design 3)

  • Patrón BeerCSS nativo: Usa `.field.label.border` para floating labels
  • Orden correcto: Input ANTES del label (requerido para animación)
  • Placeholder espacial: Usa `placeholder=" "` (un espacio, no vacío)
  • Estados: Agrega clase `.info`, `.warning`, `.valid`, o `.invalid`
  • Helper text: Usa `<span class="helper">` para todo tipo de mensajes
  • Focus state: Automáticamente usa el color púrpura Khipu (#8347AD)
  • Accesibilidad: Siempre conectar label con atributo `for`

Arrastra tu archivo aquí o

<div class="kds-file-upload">
    <div class="kds-file-upload-zone" id="fileZone"
         ondrop="handleDrop(event)" ondragover="handleDragOver(event)" ondragleave="handleDragLeave(event)">
        <i class="material-symbols-outlined icon-xl icon-color-primary">upload_file</i>
        <div class="kds-file-upload-text">
            Arrastra tu archivo aquí o
            <label for="fileInput" class="kds-file-upload-button">selecciona un archivo</label>
            <input type="file" id="fileInput" accept=".pdf,.jpg,.jpeg,.png"
                   onchange="handleFileSelect(event)" class="hidden">
        </div>
        <span class="kds-file-upload-helper">PDF, JPG o PNG (máx. 5MB)</span>
    </div>
    <div class="kds-file-upload-preview" id="filePreview"></div>
</div>

<script>
function handleDragOver(event) {
    event.preventDefault();
    event.currentTarget.classList.add('drag-over');
}

function handleDragLeave(event) {
    event.currentTarget.classList.remove('drag-over');
}

function handleDrop(event) {
    event.preventDefault();
    event.currentTarget.classList.remove('drag-over');
    const files = event.dataTransfer.files;
    if (files.length > 0) {
        displayFile(files[0]);
    }
}

function handleFileSelect(event) {
    const files = event.target.files;
    if (files.length > 0) {
        displayFile(files[0]);
    }
}

function displayFile(file) {
    const preview = document.getElementById('filePreview');
    const maxSize = 5 * 1024 * 1024; // 5MB

    // Validate size
    if (file.size > maxSize) {
        alert('El archivo excede el tamaño máximo de 5MB');
        return;
    }

    // Create preview
    preview.innerHTML = \`
        <div class="kds-file-item">
            <i class="material-symbols-outlined">description</i>
            <span class="file-name">\${file.name}</span>
            <span class="file-size">\${(file.size / 1024).toFixed(2)} KB</span>
            <button onclick="removeFile()" class="transparent circle">
                <i class="material-symbols-outlined">close</i>
            </button>
        </div>
    \`;
}

function removeFile() {
    document.getElementById('filePreview').innerHTML = '';
    document.getElementById('fileInput').value = '';
}
</script>

Usage Notes

  • Drag & Drop: Clase .drag-over se aplica automáticamente al arrastrar
  • Preview: Muestra nombre, tamaño y botón de eliminar
  • Validation: Valida formato (accept) y tamaño (5MB max)
  • Multiple: Agrega atributo multiple para múltiples archivos
